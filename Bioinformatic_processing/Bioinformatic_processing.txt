# Sorting with Picard #

#!/bin/bash
#$ -l qname=Big.q
#$ -cwd
#$ -e sorting.error
#$ -o sorting.log
#$ -N sorting.gra

for f in $(ls /data2/BSyBS/esotocor/ddradseq/reads/granulosum/trimmed_files/BAM/*.bam)
do
        picard SortSam -I $f -O $"`basename $f _paired.bam`_sorted.bam" -SO=coordinate
done

# Removing duplicated reads with Picard

#!/bin/bash
for f in $(ls /home/esotocor/Doctorado/pilotos_raws/Analisis_Submuestras/8M/*.sorted_8.bam)
do
        picard MarkDuplicates ASSUME_SORTED=true I=$f O="`basename $f .cleaned.bam`.rmdp.bam" M=marked_dup_metrics.txt
done

#Removing secondary reads with Samtools

#!/bin/bash

for f in $(ls /home/esotocor/Doctorado/pilotos_raws/Analisis_Submuestras/8M/*.rmdp.bam)
do
        samtools view -b -F3332 $f > "`basename $f .sr.rmdp.bam.sorted_8.bam`.cleaned_8.bam"

done

#Obtaining a chromosome map
bcftools view -H alt_fb_org_nomissing_bialeles_maf0.05_hwe0.0001.recode.vcf | cut -f 1 | uniq | awk '{print $0"\t"$0}' > chrom-map.txt
#Convert vcf to ped, map file

#R2 prunning
./plink --file alt.variantplink --allow-extra-chr --indep-pairwise 100 5 0.5

# Extract the the file with snps pruned from ped file
./plink --file alt.variantplink --extract plink.prune.in --allow-extra-chr --make-bed --out alt.variants.pruned

#Convert the bed file to ped file and generate map file
./plink --bfile alt.variants.pruned --allow-extra-chr --recode tab --out alt.variants.pruned
